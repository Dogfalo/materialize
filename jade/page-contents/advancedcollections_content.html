Collections allow you to group list objects together.

Basic


<div class="row">
    <div class="col s12">
        <div id="list1"></div>
    </div>
</div>
            

$(document).ready(function(){
    var list1 = new DataList("t",
              [{t: "Sam"},{t: "Allissa"}, {t: "Deb"}, {t: "Mike"}, {t: "Eli"}]
              ).setName("list1");

    $("#list1").replaceWith(list1.renderToHTML());
});
